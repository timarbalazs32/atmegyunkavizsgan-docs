Backend Dokumentáció

Ezen az oldalon a Node.js alapú backend fejlesztéshez kapcsolódó adatbázis-kezelési megoldásokat találod meg, különös tekintettel az SQLite3 variánsokra.

1. SQLite3 minta (Aszinkron / Callback alapú)

Az alapvető sqlite3 driver használata Express.js környezetben. Ez a könyvtár aszinkron módon, callback függvények segítségével kezeli a kéréseket, és modern import (ESM) szintaxist használ.

index.js (sqlite3) 
import express from 'express';
import cors from 'cors';
import sqlite3 from 'sqlite3';

const app = express();
const port = 8000;

app.use(cors());
app.use(express.json());

// Adatbázis kapcsolat létrehozása (sakila.db fájl)
const db = new sqlite3.Database(`sakila.db`);

// Alapértelmezett üdvözlő útvonal
app.get('/', (req, res) => {
    res.json({ message: 'Welcome to the Film Portal Backend' });
});

// Összes film lekérdezése (lapozással / pagination)
app.get(`/films`, (req, res) => {
    try {
        let { limit, offset } = req.query;

        // Alapértelmezett értékek, ha nincsenek megadva paraméterek
        if (!limit || !offset) {
            limit = 50;
            offset = 0;
        }

        if (typeof limit !== "number" || typeof offset !== "number") {
            return res.status(400).json({ error: "The parameters have invalid datatypes" });
        }

        const sql = `select * from film`;

        // db.all: lekéri az összes sort a megadott SQL alapján
        db.all(sql, [], function(error, rows) {
            if (error) {
                return res.status(400).json({ error: error.message });
            }
            return res.status(200).json(rows);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }
});

// Egy film lekérdezése ID alapján
app.get(`/films/:id`, (req, res) => {
    try {
        const { id } = req.params;

        if (!id) {
            return res.status(400).json({ message: `invalid parameter` });
        }

        // Biztonságos paraméterezett query a SQL injection ellen (?)
        const sql = `select * from film where film_id = ?`;

        db.all(sql, [parseInt(id)], function(error, rows) {
            if (error) {
                return res.status(404).json({ error: error.message });
            }

            if (rows.length > 0) {
                return res.status(200).json(rows);
            } else {
                return res.status(404).json({ error: "Nincs ilyen film." });
            }
        });
    } catch (error) {
        return res.status(404).json({ error: error.message });
    }
});

// Új film hozzáadása (POST kérés)
app.post(`/films`, (req, res) => {
    try {
        const {
            film_id, title, description, release_year, language_id, 
            rental_duration, rental_rate, length, replacement_cost, 
            rating, special_features
        } = req.body;

        // Kötelező mezők meglétének ellenőrzése
        if (!film_id || !title || !description || !release_year || !language_id || 
            !rental_duration || !rental_rate || !length || !replacement_cost || !rating || !special_features) {
            return res.status(400).json({ error: `Egy kötelező paraméter hiányzik.` });
        }

        let date = new Date().toISOString();

        const sql = `insert into film (film_id, title, description, release_year, language_id, rental_duration, rental_rate, length,
            replacement_cost, rating, special_features, last_update) values(?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?);`;

        // db.run: Adatmódosító (INSERT, UPDATE, DELETE) műveletekhez használatos
        db.run(sql, [film_id, title, description, release_year, language_id, rental_duration, rental_rate, length, replacement_cost, rating, special_features, date], function(error) {
            if (error) {
                return res.status(400).json({ error: error.message });
            }
            // this.lastID visszaadja a frissen generált rekord azonosítóját
            return res.status(201).json({ id: this.lastID });
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }
});

// Kategória alapú szűrés
app.get(`/category-films/:category_id`, (req, res) => {
    try {
        const { category_id } = req.params;
        const category_id_num = Number(category_id);
        
        const sql_getCategory = `select film.* from film join film_category on film.film_id = film_category.film_id where film_category.category_id = ?`;
        
        db.all(sql_getCategory, [category_id_num], function(error, rows) {
            if (error) {
                return res.status(400).json({ error: error.message });
            }
            return res.status(200).json(rows);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }
});

// Színészek listázása
app.get(`/actors`, (req, res) => {

// Kategóriák listázása
app.get(`/categories`, (req, res) => {
    try {

        db.all(sql, [], function(error, rows) {
            if (error) {
                return res.status(400).json({ error: error.message });

            return res.status(200).json(rows);
        });
    } catch (error) {

    }

// Vásárlók lekérése
app.get(`/customers`, (req, res) => {
    try {
        const sql = `select * from customer`;
        db.all(sql, [], function(error, rows) {
            if (error) {
                return res.status(400).json({ error: error.message });
            }
            return res.status(200).json(rows);
        });
    } catch (error) {

// Nyelvek lekérése
app.get(`/languages`, (req, res) => {
    try {

        db.all(sql, [], function(error, rows) {
            if (error) {

            }
            return res.status(200).json(rows);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }

// Adott filmben játszó színészek lekérése JOIN segítségével
app.get(`/film-actors/:id`, (req, res) => {
    try {
        const { id } = req.params;
        const num_id = Number(id);

        const sql_getFilm = `select actor.first_name FROM actor JOIN film_actor ON actor.actor_id = film_actor.actor_id WHERE film_actor.film_id = ?`;

        db.all(sql_getFilm, [num_id], function(error, rows) {
            if (error) {
                return res.status(400).json({ error: error.message });
            }
            return res.status(200).json(rows);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });

// Új színész rögzítése
app.post(`/actors`, (req, res) => {
    try {
        const { actor_id, first_name, last_name } = req.body;

        if (!actor_id || !first_name || !last_name) {
            return res.status(400).json({ message: "Hiányos paraméterek" });
        }

        const sql = `insert into actor (actor_id, first_name, last_name, last_update) values(?, ?, ?, ?)`;
        let date = new Date().toISOString();

        db.run(sql, [actor_id, first_name, last_name, date], function(error) {
            if (error) {
                return res.status(400).json({ message: error.message });
            }
            return res.status(201).json(this.lastID);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }
});

// Vásárló kölcsönzéseinek lekérése
app.get(`/customer-rentals/:customer_id`, (req, res) => {
    try {
        const { customer_id } = req.params;

        if (!customer_id) {
            return res.status(400).json({ message: "Egy kötelező paraméter hiányzik." });
        }

        const sql = `select rental_id, rental_date, return_date from rental where customer_id = ?`;

        db.all(sql, [customer_id], function(error, rows) {
            if (error) {
                return res.status(400).json(error.message);
            }
            return res.status(200).json(rows);
        });
    } catch (error) {
        return res.status(400).json({ error: error.message });
    }
});

app.listen(port, () => {
    console.log(`Server is running on port ${port}`);
});

2. Better-SQLite3 minta (Szinkron / Mozi hálózat kezelő API)

Ez a példa a better-sqlite3 csomagot használja szinkron módon egy mozinézési és filmnyilvántartó rendszerhez. A projekt hagyományos CommonJS (require) modulfelkéréssel épül fel.

index.js (better-sqlite3) 
const cors = require('cors');
const express = require('express');
const Database = require('better-sqlite3');

const app = express();
app.use(express.json());

// TODO 1: CORS engedélyezése (cors csomag)
app.use(cors());

// TODO 3: mozihalo.db adatbázis használata
const db = new Database('mozihalo.db');

// TODO 5: GET / – üdvözlő JSON üzenet
app.get("/", (req, res) => {
  res.json({ message: "Üdvözlés" });
});

// TODO 6: GET /filmek - Az összes film lekérdezése
app.get("/filmek", (req, res) => {
  try {
    const sql = "select * from filmek";
    const result = db.prepare(sql).all(); // .all() az összes találatot visszaadja tömbként
    return res.status(200).json(result);
  }
  catch (error) {
    return res.status(500).json(error.message);
  }
});

// TODO 7: GET /vetites/:vazon (JOIN: fcim, mnev, varos) - Vetítés adatai komplex lekérdezéssel
app.get("/vetites/:vazon", (req, res) => {
  try {
    const vazon = req.params.vazon;
    const sql = `SELECT filmek.fcim, mozik.mnev, mozik.varos 
                 FROM vetitesek 
                 INNER JOIN mozik ON mozik.mazon = vetitesek.mazon 
                 INNER JOIN filmek ON filmek.fazon = vetitesek.fazon 
                 WHERE vetitesek.vazon = ?`;
    
    const result = db.prepare(sql).get(Number(vazon)); // .get() csak egyetlen objektumot ad vissza
    return res.status(200).json(result);
  }
  catch (error) {
    return res.status(500).json(error.message);
  }
});

// TODO 8: POST /filmek - Új film felvétele a rendszerbe
app.post("/filmek", (req, res) => {
  try {
    const { fazon, fcim, rendezo, gyartasi_ev, mufaj, jegypont } = req.body;
    const sql = "insert into filmek (fazon, fcim, rendezo, gyartasi_ev, mufaj, jegypont) values (?, ?, ?, ?, ?, ?)";
    
    const result = db.prepare(sql).run(fazon, fcim, rendezo, gyartasi_ev, mufaj, jegypont); // .run() adatmódosításra
    return res.status(201).json({ message: "Film sikeresen hozzáadva", id: result.lastInsertRowid });
  }
  catch (error) {
    return res.status(500).json(error.message);
  }
});

// TODO 9: DELETE /film/:fazon - Film törlése azonosító alapján
app.delete("/film/:fazon", (req, res) => {
  try {
    const fazon = Number(req.params.fazon);
    const sql = "delete from filmek where fazon = ?";
    db.prepare(sql).run(fazon);
    return res.status(200).json({ message: "Film sikeresen törölve" });
  }
  catch (error) {
    return res.status(500).json(error.message);
  }
});

// TODO 10: POST /mozik - Új mozi helyszín rögzítése
app.post("/mozik", (req, res) => {
  const { mazon, mnev, varos, termek_szama } = req.body;
  try {
    const sql = "insert into mozik (mazon, mnev, varos, termek_szama) values (?, ?, ?, ?)";
    const result = db.prepare(sql).run(mazon, mnev, varos, termek_szama);
    return res.status(201).json({ message: "Mozi sikeresen hozzáadva", id: result.lastInsertRowid });
  }
  catch (error) {
    return res.status(500).json(error.message);
  }
});

// TODO 11: PATCH /vetites/:vazon - Vetítési adatok részleges frissítése
app.patch("/vetites/:vazon", (req, res) => {
  try {
    const vazon = Number(req.params.vazon);
    const { fazon, mazon, kezdete } = req.body;
    const sql = "update vetitesek set fazon = ?, mazon = ?, kezdete = ? where vazon = ?";
    db.prepare(sql).run(fazon, mazon, kezdete, vazon);
    return res.status(200).json({ message: "Vetítés sikeresen frissítve" });
  }
  catch (error)
  {
    return res.status(500).json(error.message);
  }
});

// TODO 2: 8000-es port konfigurálása és szerver indítása
const PORT = 8000;
app.listen(PORT, () => {
  console.log(`Szerver fut: http://localhost:${PORT}`);
});
