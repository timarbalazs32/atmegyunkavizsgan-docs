1. SQLite3 minta (Aszinkron / Callback alapú)

index.js (sqlite3) 
import express from 'express';
import cors from 'cors';
import sqlite3 from 'sqlite3';

const app = express();
app.use(cors());
app.use(express.json());

const db = new sqlite3.Database(`sakila.db`);
app.listen(8000);

6. Adatfelvételi Űrlap (Template-driven Form)

tanulo-form.ts 
import { Component } from '@angular/core';
import { FormsModule } from '@angular/forms';
import { TanuloService } from '../../services/tanulo-service';

@Component({
  selector: 'app-tanulo-form',
  standalone: true,
  imports: [FormsModule],
  templateUrl: './tanulo-form.html',
})
export class TanuloForm {
  constructor(private TanuloService: TanuloService) {}

  onSubmit(form: any) {
    this.TanuloService.addTanulo(form.value).subscribe({
      next: (response) => { form.reset(); },
      error: (err) => { console.error(err); }
    });
  }
}

7. Útvonalkezelés (Routing / Client-side Navigation)

Az Angular Routing biztosítja, hogy az alkalmazás egyoldalas (SPA) maradjon. Ahelyett, hogy a szerverről új HTML fájlokat kérne a böngésző, az Angular egyszerűen kicseréli a komponenseket a képernyőn a <router-outlet> helyén.

A) Útvonalak regisztrálása (app.routes.ts)

Itt határozzuk meg, hogy a böngésző címsorában lévő URL-hez melyik komponens tartozzon. Vizsgatipp: A sorrend számít! A konkrét útvonalak legyenek felül, az aszteriszk (vadkártya) pedig a legalsó sorban.

app.routes.ts 
import { Routes } from '@angular/router';
import { Main } from './components/main/main';
import { Tanulok } from './components/tanulok/tanulok';
import { TanuloForm } from './components/tanulo-form/tanulo-form';

export const routes: Routes = [
  // Ha a címsor üres (http://localhost:4200), a Main komponens tölt be
  { path: '', component: Main },

  // Ha a címsor /tanulok, a listázó komponens jelenik meg
  { path: 'tanulok', component: Tanulok },

  // Ha a címsor /tanuloForm, az adatfelvételi űrlap jelenik meg
  { path: 'tanuloForm', component: TanuloForm },

  // Vadkártya (Wildcard) útvonal: Ha a felhasználó beír valami hülyeséget, ami nem egyezik a fentiekkel, 
  // akkor automatikusan visszadobjuk (átirányítjuk) a főoldalra. KÖTELEZŐ ELEM A VIZSGÁN!
  { path: '**', redirectTo: '' }
];

B) Menü/Navigációs Komponens logikája (menu.ts)

Fontos elmélet: Ahhoz, hogy egy komponensben működjön a belső navigáció, az `imports` tömbbe be kell emelni a RouterOutlet (a helyőrző a betöltendő komponenseknek) és a RouterLink (a kattintható linkek kezelője) direktívákat!

menu.ts 
import { Component } from '@angular/core';
import { RouterOutlet, RouterLink } from '@angular/router'; // <-- KÖTELEZŐ importok!

@Component({
  standalone: true,
  selector: 'app-menu',
  // Be kell regisztrálni őket, hogy a HTML sablon felismerje a [routerLink] és  elemeket
  imports: [RouterOutlet, RouterLink], 
  templateUrl: './menu.html',
  styleUrl: './menu.css',
})
export class Menu {
  // Itt most nincs szükség logikára, a komponens strukturális feladatot lát el (Navbár + Helyőrző)
}
